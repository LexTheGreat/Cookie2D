protected override void Initialize()
        {
            spriteBatch = new SpriteBatch();
            
            WindowControl window = new WindowControl(GUI, "Main Menu");
            window.SetSize(400, 250);
            window.SetPosition(200, 250);
            window.DisableResizing();
            window.IsClosable = false;

            Label lblUsername = new Label(window);
            lblUsername.Text = "Username:";
            lblUsername.SetPosition(40, 30);

            Label lblPassword = new Label(window);
            lblPassword.Text = "Password:";
            lblPassword.SetPosition(lblUsername.X, lblUsername.Y + 40);

            TextBox txtUsername = new TextBox(window);
            txtUsername.SetPosition(100, 25);
            txtUsername.SetSize(250, 25);

            TextBoxPassword txtPassword = new TextBoxPassword(window);
            txtPassword.SetPosition(txtUsername.X, txtUsername.Y + 40);
            txtPassword.SetSize(txtUsername.Width, txtUsername.Height);

            Button btnLogin = new Button(window);
            btnLogin.Text = "Login";
            btnLogin.SetPosition(40, 150);
            btnLogin.SetSize(100, 40);
            btnLogin.SetImage("Content/textures/gui/buttons/login.png");

            Button btnReg = new Button(window);
            btnReg.Text = "Register";
            btnReg.SetPosition(btnLogin.X + btnLogin.Width + 5, btnLogin.Y);
            btnReg.SetSize(btnLogin.Width, btnLogin.Height);
            btnReg.SetImage("Content/textures/gui/buttons/register.png");
            
            Button btnCredits = new Button(window);
            btnCredits.Text = "Credits";
            btnCredits.SetPosition(btnReg.X + btnReg.Width + 5, btnLogin.Y);
            btnCredits.SetSize(btnLogin.Width, btnLogin.Height);
            btnCredits.SetImage("Content/textures/gui/buttons/credits.png");

            _console = new Windows.Console(GUI);
        }
